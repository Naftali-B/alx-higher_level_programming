import unittest

class TestMyList(unittest.TestCase):
    def test_print_sorted_empty_list(self):
        my_list = MyList()
        expected_output = []
        my_list.print_sorted()
        self.assertEqual(my_list, expected_output)

    def test_print_sorted_sorted_list(self):
        my_list = MyList([1, 2, 3, 4, 5])
        expected_output = [1, 2, 3, 4, 5]
        my_list.print_sorted()
        self.assertEqual(my_list, expected_output)

    def test_print_sorted_unsorted_list(self):
        my_list = MyList([5, 3, 1, 4, 2])
        expected_output = [1, 2, 3, 4, 5]
        my_list.print_sorted()
        self.assertEqual(my_list, expected_output)

    def test_print_sorted_mixed_list(self):
        my_list = MyList([5, 'apple', 3, 'banana', 1])
        expected_output = ['apple', 1, 3, 'banana', 5]
        my_list.print_sorted()
        self.assertEqual(my_list, expected_output)

if __name__ == '__main__':
    unittest.main()
