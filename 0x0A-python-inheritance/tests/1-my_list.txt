import unittest
from .. import MyList

class TestMyList(unittest.TestCase):
    def test_basic_functionality(self):
        my_list = MyList()
        my_list.append(1)
        my_list.append(4)
        my_list.append(2)
        my_list.append(3)
        my_list.append(5)

        self.assertEqual(my_list, [1, 4, 2, 3, 5])
        my_list.print_sorted()
        self.assertEqual(my_list, [1, 4, 2, 3, 5])

    def test_empty_list(self):
        empty_list = MyList()
        empty_list.print_sorted()
        self.assertEqual(empty_list, [])

    def test_repeated_elements(self):
        my_list = MyList()
        my_list.append(3)
        my_list.append(1)
        my_list.append(3)

        self.assertEqual(my_list, [3, 1, 3])
        my_list.print_sorted()
        self.assertEqual(my_list, [1, 3, 3])

    def test_negative_integers(self):
        my_list = MyList()
        my_list.append(-2)
        my_list.append(5)
        my_list.append(-10)
        my_list.append(-1)

        self.assertEqual(my_list, [-2, 5, -10, -1])
        my_list.print_sorted()
        self.assertEqual(my_list, [-10, -2, -1, 5])

    def test_large_list(self):
        my_list = MyList()
        for i in range(1000, 0, -1):
            my_list.append(i)

        my_list.print_sorted()
        self.assertEqual(my_list, list(range(1, 1001)))

if __name__ == '__main__':
    unittest.main()
